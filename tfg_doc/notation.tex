\chapter*{Notación}
\thispagestyle{empty}

\begin{description}
  \item[$P$] Potencia disipada estimada [mW]
  \item[$c_i$] Coeficiente de potencia de la categoría $i$ [mW/instrucción]
  \item[$n_i$] Número de instrucciones ejecutadas de la categoría $i$
  \item[$N$] Número de categorías de instrucción (6)
  \item[$V_{DD}$] Tensión de alimentación [V]
  \item[$I$] Corriente de alimentación [A]
  \item[$R_{\text{shunt}}$] Resistencia de derivación para medición de corriente [$\Omega$]
  \item[$G$] Ganancia del amplificador de sensado de corriente INA240
  \item[$V_{\text{out}}$] Tensión de salida del INA240 [mV]
  \item[CSR] \textit{Control and Status Register} — registro de control y estado RISC-V
  \item[DTM] \textit{Debug Transport Module} — módulo de transporte de depuración
  \item[ISA] \textit{Instruction Set Architecture} — arquitectura del conjunto de instrucciones
  \item[RTL] \textit{Register Transfer Level} — nivel de transferencia entre registros
  \item[SoC] \textit{System-on-Chip} — sistema en un chip
\end{description}
